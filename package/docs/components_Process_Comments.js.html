import React from "react";
import Table from "src/components/ui/Table";

/**
 * Generic Comments History Component
 * Displays a table of comments with role-based badges
 * 
 * @param {Object} props - Component props
 * @param {Array} props.comments - Array of comment objects
 * @param {boolean} props.loading - Loading state
 * @param {string} props.title - Section title (default: "Comment & Approval History")
 * @param {Object} props.customStyles - Optional custom styles override
 * @param {Function} props.onRefresh - Optional refresh callback
 * @param {string} props.emptyMessage - Custom message when no comments (default: "No comments found.")
 * @param {Object} props.roleColors - Custom role color mapping
 * 
 * Comment object structure:
 * {
 *   id: number,
 *   commenter: string,
 *   role: string,
 *   comment: string,
 *   date: string
 * }
 */
const CommentsHistory = ({
  comments = [],
  loading = false,
  title = "Comment & Approval History",
  customStyles = {},
  onRefresh = null,
  emptyMessage = "No comments found.",
  roleColors = {},
}) => {
  const defaultRoleColors = {
    Initiator: { bg: "#DBEAFE", text: "#1E40AF" },
    Approver: { bg: "#FEF3C7", text: "#92400E" },
    Reviewer: { bg: "#E0E7FF", text: "#3730A3" },
    Admin: { bg: "#F3E8FF", text: "#6B21A8" },
    Default: { bg: "#F3F4F6", text: "#374151" },
  };

  const mergedRoleColors = { ...defaultRoleColors, ...roleColors };
  const getRoleColor = (role) => {
    return mergedRoleColors[role] || mergedRoleColors.Default;
  };

  const styles = {
    container: {
      marginBottom: "24px",
      padding: "24px",
      border: "1px solid #E5E7EB",
      borderRadius: "10px",
      backgroundColor: "#F9FAFB",
      boxShadow: "0 0 5px rgba(0, 0, 0, 0.03)",
      ...customStyles.container,
    },
    header: {
      display: "flex",
      alignItems: "center",
      justifyContent: "space-between",
      marginBottom: "20px",
      borderBottom: "1px solid #CFD8DC",
      paddingBottom: "12px",
      ...customStyles.header,
    },
    heading: {
      margin: 0,
      fontSize: "18px",
      fontWeight: "700",
      color: "#3F51B5",
      letterSpacing: "0.3px",
      ...customStyles.heading,
    },
    refreshButton: {
      padding: "6px 12px",
      border: "1px solid #3F51B5",
      borderRadius: "6px",
      backgroundColor: "transparent",
      color: "#3F51B5",
      cursor: "pointer",
      fontSize: "13px",
      fontWeight: "600",
      transition: "all 0.3s ease",
      ...customStyles.refreshButton,
    },
    tableWrapper: {
      overflowX: "auto",
      ...customStyles.tableWrapper,
    },
    badge: {
      padding: "4px 8px",
      borderRadius: "4px",
      fontSize: "12px",
      fontWeight: "500",
      display: "inline-block",
      ...customStyles.badge,
    },
    loadingText: {
      textAlign: "center",
      padding: "20px",
      color: "#6B7280",
      fontSize: "14px",
      ...customStyles.loadingText,
    },
    emptyText: {
      textAlign: "center",
      padding: "20px",
      color: "#6B7280",
      fontSize: "14px",
      ...customStyles.emptyText,
    },
  };

  const columns = [
    {
      key: "id",
      label: "SI.No",
    },
    {
      key: "commenter",
      label: "Commenter",
    },
    {
      key: "role",
      label: "Role",
      render: (role) => {
        const roleColor = getRoleColor(role);
        return (
          <span
            style={{
              ...styles.badge,
              backgroundColor: roleColor.bg,
              color: roleColor.text,
            }}
          >
            {role}
          </span>
        );
      },
    },
    {
      key: "comment",
      label: "Comment",
    },
    {
      key: "date",
      label: "Date",
    },
  ];

  return (
    <div style={styles.container}>
      <div style={styles.header}>
        <h3 style={styles.heading}>{title}</h3>
        {onRefresh && (
          <button
            type="button"
            onClick={onRefresh}
            style={styles.refreshButton}
            onMouseEnter={(e) => {
              e.target.style.backgroundColor = "#3F51B5";
              e.target.style.color = "#FFFFFF";
            }}
            onMouseLeave={(e) => {
              e.target.style.backgroundColor = "transparent";
              e.target.style.color = "#3F51B5";
            }}
          >
            ↻ Refresh
          </button>
        )}
      </div>

      {loading && comments.length === 0 ? (
        <div style={styles.loadingText}>Loading comments...</div>
      ) : (
        <div style={styles.tableWrapper}>
          {comments.length > 0 ? (
            <Table columns={columns} data={comments} />
          ) : (
            <div style={styles.emptyText}>{emptyMessage}</div>
          )}
        </div>
      )}
    </div>
  );
};

export default CommentsHistory;
