import React from "react";

/**
 * @fileoverview BadgeChip component.
 * Displays a small pill-shaped badge with a text label.
 * Useful for statuses like "In Progress" or "Completed".
 * @module src/components/ui
 * @date 12/09/2025
 * @since 1.0.0
 */

/**
 * BadgeChip Component
 * Renders a badge with customizable background and text color.
 *
 * ðŸ“ Example:
 * Input:
 * <BadgeChip text="In Progress" color="#2563EB" />
 *
 * Output:
 * A pill-shaped blue badge containing the words "In Progress".
 *
 * @param {Object} props
 * @param {string} props.text - The text inside the badge.
 * @param {string} [props.color="#6B7280"] - Background color of the badge.
 * @param {string} [props.textColor="white"] - Text color inside the badge.
 * @returns {JSX.Element} Badge chip component.
 */
const BadgeChip = ({ text, color = "#6B7280", textColor = "white" }) => {
  return (
    <span
      style={{
        display: "inline-block",
        padding: "4px 10px",
        borderRadius: "9999px",
        fontSize: "12px",
        fontWeight: "600",
        backgroundColor: color,
        color: textColor,
        whiteSpace: "nowrap",
      }}
    >
      {text}
    </span>
  );
};

export default BadgeChip;
