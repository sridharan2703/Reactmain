/**
 * @fileoverview
 * This file defines a small reusable section component for forms.
 * It visually separates parts of a form using a colorful top border.
 * 
 * ðŸ§© Example:
 * ```
 * <CardStyle>
 *   <p>This is one part of the form.</p>
 * </CardStyle>
 * ```
 * Every time the page loads, the top border will have a random color.
 */

import React, { useState, useEffect } from "react";

/**
 * CardStyle Component
 *
 * A simple container that wraps form content with padding, a light background,
 * and a colorful top border chosen randomly.
 *
 * @component
 * @param {Object} props - The properties passed into this component.
 * @param {React.ReactNode} props.children - The elements (like text boxes, labels, etc.) that go inside the form section.
 * @param {Object} [props.style={}] - Optional custom styles to change the appearance.
 *
 * @returns {JSX.Element} A styled box that visually groups part of a form.
 */
const CardStyle = ({ children, style = {} }) => {
  /**
   * This state variable stores the color of the top border.
   * It starts with a default green color (#10B981).
   */
  const [borderColor, setBorderColor] = useState("#10B981");

  /**
   * useEffect Hook:
   * Runs once when the component first appears on the screen.
   * It picks one random color from the list below and sets it as the border color.
   */
  useEffect(() => {
    const colors = [
      "#dbbf21ff", // yellow
      "#0000FF",   // blue
      "#FFA500",   // orange
      "#640f64ff", // magenta (purple shade)
      "#FF0000",   // red
    ];

    // Choose one random color from the list
    const randomColor = colors[Math.floor(Math.random() * colors.length)];

    // Update the border color with the chosen color
    setBorderColor(randomColor);
  }, []);

  /**
   * Default visual styling for the form section.
   * You can think of this like decorating a box with padding and rounded corners.
   */
  const defaultStyle = {
    marginBottom: "30px",        // space below each section
    padding: "20px",             // space inside the section
    border: "1px solid #e5e7eb", // light gray border around the box
    borderTop: `4px solid ${borderColor}`, // colorful top border
    borderRadius: "10px",        // smooth rounded corners
    backgroundColor: "#ffffff",  // white background
  };

  // The component returns a "div" (a box in HTML) with the styles applied.
  return <div style={{ ...defaultStyle, ...style }}>{children}</div>;
};

export default CardStyle;
