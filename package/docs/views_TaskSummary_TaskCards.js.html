import React, { useState } from 'react';
import { taskCardStyles as styles } from 'src/components/ui/StatCards.js';

/**
 * A UI component that displays a summary of a single task in a card format.
 * It shows key information and is clickable to view more details.
 *
 * @component
 * @param {object} props The component props.
 * @param {object} props.task The task object containing details to display.
 * @param {string} props.task.id The unique identifier for the task.
 * @param {string} props.task.title The title of the task.
 * @param {('high'|'medium'|'low')} props.task.priority The priority level of the task.
 * @param {string} props.task.status The current status of the task (e.g., "Completed", "Pending").
 * @param {string} props.task.borderColor A color hex code for the card's left border.
 * @param {string} props.task.bgColor A color hex code for the status tag's background.
 * @param {string} props.task.statusColor A color hex code for the status tag's text.
 * @param {object} props.task.assignee An object with details of the person assigned to the task.
 * @param {string} props.task.assignee.name The name of the assignee.
 * @param {string} props.task.assignee.role The role of the assignee.
 * @param {string} props.task.assignee.department The department of the assignee.
 * @param {string} props.task.elapsed The time elapsed since the task was initiated.
 * @param {function(string): React.CSSProperties} props.getPriorityStyles A function that returns style objects based on the priority level.
 * @param {function(object): void} props.openDialog A callback function to open the details dialog, passing the task object.
 * @returns {JSX.Element} The rendered TaskCard component.
 */
const TaskCard = ({ task, getPriorityStyles, openDialog }) => {
  const [isHovered, setIsHovered] = useState(false);
  const containerDynamicStyle = {
    ...styles.container,
    borderLeft: `5px solid ${task.borderColor}`,
    boxShadow: isHovered ? '0 10px 20px rgba(0,0,0,0.1)' : '0 2px 8px rgba(0,0,0,0.05)',
    transform: isHovered ? 'translateY(-5px)' : 'translateY(0)',
  };

  const priorityTagStyle = { ...styles.tag, ...getPriorityStyles(task.priority) };
  const statusTagStyle = { ...styles.tag, backgroundColor: task.bgColor, color: task.statusColor };
  const avatarStyle = { ...styles.avatar, backgroundColor: task.borderColor };

  return (
    <div
      style={containerDynamicStyle}
      onMouseEnter={() => setIsHovered(true)}
      onMouseLeave={() => setIsHovered(false)}
      onClick={() => openDialog(task)}
    >
      <div>
        <div style={styles.header}>
          <h3 style={styles.title}>{task.title}</h3>
          <span style={styles.taskId}>Task ID: {task.id}/V</span>
        </div>

        <div style={styles.tagsContainer}>
          <span style={priorityTagStyle}>{task.priority}</span>
          <span style={statusTagStyle}>{task.status}</span>
        </div>

        <div style={styles.assigneeContainer}>
          <div style={avatarStyle}>
             <img src="/placeholder-avatar.jpg" alt={task.assignee.name} style={styles.avatarImage} onError={(e) => { e.target.onerror = null; e.target.style.display='none'; e.target.parentNode.innerHTML = task.assignee.name.split(' ').map(n => n[0]).join(''); }}/>
          </div>
          <div>
            <div style={styles.assigneeInfoName}>{task.assignee.name}</div>
            <div style={styles.assigneeInfoRole}>{task.assignee.role}, {task.assignee.department}</div>
          </div>
        </div>

        <div style={styles.footer}>
          <svg xmlns="http://www.w3.org/2000/svg" width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" style={styles.icon}>
            <circle cx="9" cy="9" r="8"></circle>
            <polyline points="12 6 12 12 16 14"></polyline>
          </svg>
          <span>Elapsed:</span>
          <span style={styles.elapsedTime}>{task.elapsed}</span>
        </div>
      </div>
    </div>
  );
};

export default TaskCard;
