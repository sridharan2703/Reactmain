/**

const TaskSummaryPage = () => {
  const [tasks, setTasks] = useState([]);
  const [filteredTasks, setFilteredTasks] = useState([]);
  const [currentPage, setCurrentPage] = useState(1);
  const [isDialogOpen, setIsDialogOpen] = useState(false);
  const [selectedTask, setSelectedTask] = useState(null);
  
  // Filter states
  const [filters, setFilters] = useState({
    search: '',
    status: 'In Progress',
    priority: 'all',
    department: 'all',
    rating: 'all',
    sortBy: 'initiated',
    sortOrder: 'desc'
  });

  useEffect(() => {
    setTasks(tasksData);
    setFilteredTasks(tasksData);
  }, []);

  // Apply filters whenever tasks or filters change
  useEffect(() => {
    let filtered = [...tasks];

    // Search filter
    if (filters.search) {
      filtered = filtered.filter(task =>
        task.title.toLowerCase().includes(filters.search.toLowerCase()) ||
        task.assignee.name.toLowerCase().includes(filters.search.toLowerCase()) ||
        task.employeeId.includes(filters.search)
      );
    }

    // Status filter
    if (filters.status !== 'all') {
      filtered = filtered.filter(task => task.status === filters.status);
    }

    // Priority filter
    if (filters.priority !== 'all') {
      filtered = filtered.filter(task => task.priority === filters.priority);
    }

    // Department filter
    if (filters.department !== 'all') {
      filtered = filtered.filter(task => task.assignee.department === filters.department);
    }

    // Rating filter
    if (filters.rating !== 'all') {
      const ratingValue = parseInt(filters.rating);
      filtered = filtered.filter(task => task.rating === ratingValue);
    }

    // Sorting
    filtered.sort((a, b) => {
      let aValue, bValue;
      
      switch (filters.sortBy) {
        case 'title':
          aValue = a.title.toLowerCase();
          bValue = b.title.toLowerCase();
          break;
        case 'priority':
          const priorityOrder = { 'high': 3, 'medium': 2, 'low': 1 };
          aValue = priorityOrder[a.priority];
          bValue = priorityOrder[b.priority];
          break;
        case 'status':
          aValue = a.status.toLowerCase();
          bValue = b.status.toLowerCase();
          break;
        case 'rating':
          aValue = a.rating;
          bValue = b.rating;
          break;
        case 'elapsed':
          aValue = parseInt(a.elapsed.split(' ')[0]);
          bValue = parseInt(b.elapsed.split(' ')[0]);
          break;
        case 'initiated':
        default:
          aValue = new Date(a.initiated.split('-').reverse().join('-'));
          bValue = new Date(b.initiated.split('-').reverse().join('-'));
          break;
      }

      if (filters.sortOrder === 'asc') {
        return aValue < bValue ? -1 : aValue > bValue ? 1 : 0;
      } else {
        return aValue > bValue ? -1 : aValue < bValue ? 1 : 0;
      }
    });

    setFilteredTasks(filtered);
    setCurrentPage(1); // Reset to first page when filters change
  }, [tasks, filters]);

          <option value="all"> All Status</option>
          <option value="Completed"> Completed</option>
          <option value="Pending"> Pending</option>
          <option value="In Progress"> In Progress</option>
          <option value="On Hold"> On Hold</option>
        </select>

        <select
          value={filters.priority}
          onChange={(e) => handleFilterChange('priority', e.target.value)}
          style={pageStyles.filterSelect}
        >
          <option value="all"> All Priority</option>
          <option value="high"> High</option>
          <option value="medium"> Medium</option>
          <option value="low"> Low</option>
        </select>

        <select
          value={filters.department}
          onChange={(e) => handleFilterChange('department', e.target.value)}
          style={pageStyles.filterSelect}
        >
          <option value="all"> All Depts</option>
          {getUniqueValues('department').map(dept => (
            <option key={dept} value={dept}>{dept}</option>
          ))}
        </select>

        {/* Compact Sort */}
        <div style={pageStyles.sortContainer}>
          <select
            value={filters.sortBy}
            onChange={(e) => handleFilterChange('sortBy', e.target.value)}
            style={pageStyles.sortSelect}
          >
            <option value="initiated"> Date</option>
            <option value="title"> Title</option>
            <option value="priority"> Priority</option>
            <option value="status"> Status</option>
            <option value="elapsed"> Time</option>
          </select>
          <button
            onClick={() => handleFilterChange('sortOrder', filters.sortOrder === 'desc' ? 'asc' : 'desc')}
            style={{
              ...pageStyles.sortSelect,
              width: '40px',
              textAlign: 'center',
              fontWeight: 'bold'
            }}
            title={`Sort ${filters.sortOrder === 'desc' ? 'Ascending' : 'Descending'}`}
          >
            {filters.sortOrder === 'desc' ? '' : ''}
          </button>
        </div>

        {/* Results & Clear */}
        <div style={pageStyles.resultsInfo}>
          <span style={pageStyles.resultsCount}>
            {filteredTasks.length}/{tasks.length}
          </span>
        </div>

        <button 
          style={pageStyles.clearButton}
          onClick={clearAllFilters}
          onMouseEnter={(e) => e.target.style.transform = 'translateY(-1px)'}
          onMouseLeave={(e) => e.target.style.transform = 'translateY(0)'}
          title="Clear all filters"
        >
           Reset
        </button>
      </div>

      <main style={pageStyles.main}>
        <div style={pageStyles.grid}>
          {currentTasks.map((task) => (
            <TaskCard
              key={task.id}
              task={task}
              getPriorityStyles={getPriorityStyles}
              openDialog={openTaskDetailDialog} // Pass the dialog opener function
            />
          ))}
        </div>

        {filteredTasks.length === 0 && (
          <div style={{ 
            textAlign: 'center', 
            padding: '60px 20px', 
            color: '#6b7280',
            backgroundColor: '#f9fafb',
            borderRadius: '12px',
            border: '2px dashed #d1d5db'
          }}>
            <div style={{ fontSize: '48px', marginBottom: '16px' }}></div>
            <h3 style={{ fontSize: '18px', fontWeight: '600', marginBottom: '8px', color: '#374151' }}>No tasks found</h3>
            <p>Try adjusting your search criteria or clear the filters to see all tasks.</p>
          </div>
        )}
      </main>

     {filteredTasks.length > 0 && (
  <footer style={pageStyles.paginationContainer}>
    <Pagination
      filteredTasks={filteredTasks}
      totalPages={totalPages}
      currentPage={currentPage}
      setCurrentPage={setCurrentPage}
    
      responsive={{ isTablet: false }}
    />
  </footer>
)}


      {/* The Task Detail Dialog Component */}
      <TaskDetailDialog
        open={isDialogOpen}
        onClose={closeTaskDetailDialog}
        task={selectedTask} 
      />
    </div>
  );
};
